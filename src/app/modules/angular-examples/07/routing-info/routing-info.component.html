Different ways of passing addresses with params:

[routerLink] = "[]" : place to click
[routerLink] = "['.']" : place to click

[routerLink] = "['/angular/RoutingInfo']" : place to click
[routerLink] = "['/angular/RoutingInfo', 3]" : place to click
[routerLink] = "['/angular/RoutingInfo', 3, 'params color']" : place to click
queryString with fragment "?id=4#some place" : place to click
queryString "?id=4&color=query%20color" : place to click
[routerLink] and queryString and fragment: place to click

Destructive links:

[routerLink] = "['RoutingInfo']" : place to click
[routerLink] = "['.', 3]" : place to click
[routerLink] = "['..']" : place to click
